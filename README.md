### Calculator Pseudocode

firstNumber;
secondNumber;
operation;

user inputs number

operation press: 
    firstnumber = input; 
    operation = button pressed;
    clearOnNextPress = true;

user inputs another number:
    clearOnNextPress = false;
    display pressed number;

operation pressed:
    firstNumber = operate num1 and num2;
    secondNumber = undefined;
    operation = button pressed;
